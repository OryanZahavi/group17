        event.preventDefault(); // מונע שליחה של הטופס כברירת מחדל

        let isValid = true;

        // אימות שם פרטי
        if (!formInputs.firstName.value.trim()) {
            formInputs.firstName.classList.add('error');
            isValid = false;
        } else {
            formInputs.firstName.classList.remove('error');
        }

        // אימות שם משפחה
        if (!formInputs.lastName.value.trim()) {
            formInputs.lastName.classList.add('error');
            isValid = false;
        } else {
            formInputs.lastName.classList.remove('error');
        }

        // אימות שם משפחה
        if (!formInputs.lastName.value.trim()) {
            formInputs.lastName.classList.add('error');
            isValid = false;
        } else {
            formInputs.lastName.classList.remove('error');
        }

        // // אימות תאריך לידה
        // if (!formInputs.birthDate.value) {

        //     alert("מצטערים, ההצטרפות הינה מגיל 18 ומעלה.");
        //     isValid = false;
        // }
        //

        if (!validateEmail(formInputs.email.value)) {
            formInputs.email.classList.add('error');
            isValid = false;
        } else {
            formInputs.email.classList.remove('error');
        }

        // // אימות מספר פלאפון
        if (!validatePhone(formInputs.phone.value)) {
            formInputs.phone.classList.add('error');
            isValid = false;
        } else {
            formInputs.phone.classList.remove('error');
        }

        // אימות מספר פלאפון
        const passwordValidationResult = validatePassword(formInputs.password.value);
        if (!passwordValidationResult.valid) {
            formInputs.password.classList.add('errorMessage');
            passwordError.textContent = passwordValidationResult.errorMessage;
            isValid = false;
        } else {
            formInputs.password.classList.remove('errorMessage');
        }

        }
    });
